% Simple question bank in LaTeX style

\question{What is the capital of France?}
\option{Berlin}
\option{Madrid}
\option{Paris}
\option{Rome}
\answer{Paris}

\question{Who developed the theory of relativity?}
\option{Isaac Newton}
\option{Albert Einstein}
\option{Galileo Galilei}
\option{Niels Bohr}
\answer{Albert Einstein}

\question{Which particle mediates the electromagnetic force?}
\option{Photon}
\option{Gluon}
\option{Z boson}
\option{Higgs boson}
\answer{Photon}

\question{What is the correct commutation relation between position and momentum operators?}
\option{$[x, p] = 0$}
\option{$[x, p] = i\hbar$}
\option{$[x, p] = -i\hbar$}
\option{$[x, p] = \hbar$}
\answer{$[x, p] = i\hbar$}

\question{Which of the following quarks has the highest mass?}
\option{Charm}
\option{Bottom}
\option{Top}
\option{Strange}
\answer{Top}

\question{In Quantum Electrodynamics (QED), which vertex factor corresponds to the electron-photon interaction?}
\option{$-ie\gamma^\mu$}
\option{$-ig\gamma^\mu$}
\option{$ie\gamma^\mu$}
\option{$ig\gamma^\mu$}
\answer{$-ie\gamma^\mu$}

\question{What is the dimensionality of spacetime in special relativity?}
\option{3}
\option{2}
\option{4}
\option{5}
\answer{4}
